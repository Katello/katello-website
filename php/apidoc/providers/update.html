PUT /api/providers/:id
Update a provider

Examples

PUT /api/providers/4
{
  "provider": {
    "repository_url": "http://inecas.fedorapeople.org/fakerepos/cds/"
  }
}
200
{
  "sync_state": "not_synced",
  "name": "Red Hat",
  "id": 4,
  "repository_url": "http://inecas.fedorapeople.org/fakerepos/cds/",
  "description": null,
  "task_status_id": null,
  "created_at": "2012-08-10T13:50:25Z",
  "updated_at": "2012-08-10T13:50:27Z",
  "last_sync": null,
  "provider_type": "Red Hat",
  "organization_id": 3
}
