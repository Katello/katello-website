Create a role

POST /api/users/:user_id/roles
Create a role

Examples

POST /api/roles/
{
  "role": {
    "name": "tmp_permission_role_227e003",
    "description": null
  }
}
200
{
  "name": "tmp_permission_role_227e003",
  "created_at": "2012-08-10T13:50:06Z",
  "description": null,
  "updated_at": "2012-08-10T13:50:06Z",
  "id": 6,
  "locked": false
}

    "description": "new description of the role"
  }
}

  "description": "new description of the role",
  "updated_at": "2012-08-10T13:58:02Z",
  "id": 7,
  "locked": false
}
